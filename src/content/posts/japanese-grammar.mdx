grammarCategories:
  - category: "Day 1"
    remark: "基础判断句"
    items:
      - title: "～は～です"
        emoji: "🌱"
        level: "N5"
        levelColor: "#22c55e"
        structure: "A は B です"
        meaning: "A 是 B（判断句）"
        examples:
          - jp: "私は学生です"
            cn: "我是学生"
          - jp: "これは本です"
            cn: "这是书"
          - jp: "田中先生は日本人です"
            cn: "田中老师是日本人"
        tips: "「は」读作「wa」，是主题标记助词"

      - title: "～も"
        emoji: "🌱"
        level: "N5"
        levelColor: "#22c55e"
        structure: "A も B です"
        meaning: "A 也是 B（并列）"
        examples:
          - jp: "私も学生です"
            cn: "我也是学生"
          - jp: "これも本です"
            cn: "这也是书"
        tips: "「も」表示「也」，替代「は」使用"

      - title: "～ではありません"
        emoji: "🌱"
        level: "N5"
        levelColor: "#22c55e"
        structure: "A は B ではありません"
        meaning: "A 不是 B（否定判断）"
        examples:
          - jp: "私は学生ではありません"
            cn: "我不是学生"
          - jp: "これは本ではありません"
            cn: "这不是书"
        tips: "口语中常用「じゃないです」"

  - category: "Day 2"
    remark: "动词基础"
    items:
      - title: "～ています"
        emoji: "🌿"
        level: "N4"
        levelColor: "#3b82f6"
        structure: "动词て形 + います"
        meaning: "正在做～ / 状态的持续"
        examples:
          - jp: "食べています"
            cn: "正在吃"
          - jp: "勉強しています"
            cn: "正在学习"
          - jp: "結婚しています"
            cn: "已婚（状态）"
        tips: "某些动词表示状态持续而非正在进行"

      - title: "～たい"
        emoji: "🌿"
        level: "N4"
        levelColor: "#3b82f6"
        structure: "动词ます形（去掉ます）+ たい"
        meaning: "想要做～"
        examples:
          - jp: "食べたいです"
            cn: "想吃"
          - jp: "日本に行きたいです"
            cn: "想去日本"
        tips: "疑问形用「～たいですか」"

      - title: "～てもいいです"
        emoji: "🌿"
        level: "N4"
        levelColor: "#3b82f6"
        structure: "动词て形 + もいいです"
        meaning: "可以做～（许可）"
        examples:
          - jp: "ここに入ってもいいです"
            cn: "可以进这里"
          - jp: "写真を撮ってもいいですか"
            cn: "可以拍照吗？"
        tips: "否定许可用「～てはいけません」"

  - category: "Day 3"
    remark: "逆接表达"
    items:
      - title: "～にもかかわらず"
        emoji: "🌳"
        level: "N3"
        levelColor: "#f59e0b"
        structure: "名詞 + のにもかかわらず / 普通形 + にもかかわらず"
        meaning: "尽管～但是～"
        examples:
          - jp: "雨にもかかわらず、試合は行われた"
            cn: "尽管下雨，比赛还是举行了"
          - jp: "努力したにもかかわらず、失敗した"
            cn: "虽然努力了，但还是失败了"
        tips: "书面语，口语中常用「～のに」"

      - title: "～にもかかわらず（续）"
        emoji: "🌳"
        level: "N3"
        levelColor: "#f59e0b"
        structure: "～にもかかわらず"
        meaning: "表示与预期相反的结果"
        examples:
          - jp: "病にもかかわらず、出勤した"
            cn: "尽管生病了，还是去上班了"
          - jp: "練習したにもかかわらず、勝てなかった"
            cn: "尽管练习了，还是没能赢"
        tips: "语气比「のに」更正式"
